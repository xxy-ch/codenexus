    ```rust
    pub fn feature_gate(
        slug: &'static str,
        gateway: Arc<FeatureGatewayService>,
    ) -> impl Fn(Request, Next) -> Pin<Box<dyn Future<Output = Result<Response, StatusCode>> + Send>> + Clone {
        move |req: Request, next: Next| {
            let gateway = gateway.clone();
            let slug = slug;
            Box::pin(async move {
                let tenant_ctx = req.extensions().get::<TenantContext>()
                    .ok_or(StatusCode::UNAUTHORIZED)?;
                let resolved = gateway.resolve(slug, tenant_ctx.campus_id, tenant_ctx.grade_id).await;
                if resolved.enabled {
                    Ok(next.run(req).await)
                } else {
                    Err(StatusCode::NOT_FOUND)  // D-08: 404, not 403
                }
            })
        }
    }
    ```
