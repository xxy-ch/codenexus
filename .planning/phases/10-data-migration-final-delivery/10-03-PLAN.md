From api/migrations/005_create_problems.sql:
```sql
CREATE TABLE problems (
    id BIGSERIAL PRIMARY KEY,
    organization_id BIGINT NOT NULL REFERENCES organizations(id),
    campus_id BIGINT REFERENCES campuses(id),
    author_id UUID NOT NULL REFERENCES users(id),
    title TEXT NOT NULL, description TEXT NOT NULL,
    difficulty TEXT CHECK (difficulty IN ('easy', 'medium', 'hard')),
    visibility TEXT NOT NULL DEFAULT 'private',
    time_limit_ms INTEGER NOT NULL DEFAULT 1000,
    memory_limit_kb INTEGER NOT NULL DEFAULT 256000,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
```

From api/migrations/006_create_test_cases.sql:
```sql
CREATE TABLE test_cases (
    id BIGSERIAL PRIMARY KEY, problem_id BIGINT NOT NULL REFERENCES problems(id),
    input TEXT NOT NULL, output TEXT NOT NULL,
    is_secret BOOLEAN NOT NULL DEFAULT false, points INTEGER NOT NULL DEFAULT 1,
    order_index INTEGER NOT NULL, created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
```

From api/migrations/007_create_submissions.sql:
```sql
CREATE TABLE submissions (
    id BIGSERIAL PRIMARY KEY, organization_id BIGINT NOT NULL,
    user_id UUID NOT NULL, problem_id BIGINT NOT NULL,
    language TEXT NOT NULL CHECK (language IN ('python3', 'c', 'cpp', 'c++')),
    code TEXT NOT NULL,
    status TEXT NOT NULL DEFAULT 'queued',
    verdict TEXT CHECK (verdict IN ('ac', 'wa', 'rte', 'tle', 'mle', 'ole', 'ce', 'ie')),
    time_ms INTEGER, memory_kb INTEGER,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
```
